# Date Picker

A date picker component that allows users to select a date or date range from a calendar popup.

<DocsLink href="https://ark-ui.com/docs/components/date-picker" />

<DatePickerBasicDemo client:load />

```tsx
import {
  DatePicker,
  DatePickerLabel,
  DatePickerControl,
  DatePickerRootProvider,
  DatePickerYearSelect,
  DatePickerMonthSelect,
} from "~/components/date-picker";

// Simple usage — just 3 components
<DatePicker>
  <DatePickerLabel>Label</DatePickerLabel>
  <DatePickerControl />
</DatePicker>

// For advanced usage with custom composition, import DatePickerBase:
import { DatePickerBase } from "~/components/date-picker/date-picker.base";
```

## Installation

### CLI

Run the following command to add the component to your project:

```bash
npx solidui-cli@latest add date-picker
```

Then create `src/components/date-picker/index.tsx`:

```tsx
import {
  children,
  Show,
  splitProps,
  type Component,
} from "solid-js";
import { DatePicker as ArkDatePicker } from "@ark-ui/solid/date-picker";
import { Portal } from "solid-js/web";
import { buttonVariants, datePickerVariants } from "../../recipes/date-picker";
import { DatePickerBase } from "./date-picker.base";

const styles = datePickerVariants();

// Composite: wraps Root and includes the calendar popover
const DatePicker: Component<ArkDatePicker.RootProps> = (props) => {
  const [local, others] = splitProps(props, ["children", "class"]);
  return (
    <DatePickerBase.Root class={styles.root({ class: local.class })} {...others}>
      {local.children}
      <Portal>
        <DatePickerBase.Positioner>
          <DatePickerBase.Content>
            <DatePickerBase.View view="day">
              <DatePickerBase.Context>
                {(ctx) => (
                  <>
                    <DatePickerBase.ViewControl>
                      <DatePickerBase.PrevTrigger class={buttonVariants({ variant: "outline" })}>
                        <svg ...>...</svg>
                      </DatePickerBase.PrevTrigger>
                      <DatePickerBase.ViewTrigger class={buttonVariants({ variant: "ghost" })}>
                        <DatePickerBase.RangeText />
                      </DatePickerBase.ViewTrigger>
                      <DatePickerBase.NextTrigger class={buttonVariants({ variant: "outline" })}>
                        <svg ...>...</svg>
                      </DatePickerBase.NextTrigger>
                    </DatePickerBase.ViewControl>
                    <DatePickerBase.Table>
                      <DatePickerBase.TableHead>
                        <DatePickerBase.TableRow>
                          {ctx().weekDays.map((weekDay) => (
                            <DatePickerBase.TableHeader>{weekDay.short}</DatePickerBase.TableHeader>
                          ))}
                        </DatePickerBase.TableRow>
                      </DatePickerBase.TableHead>
                      <DatePickerBase.TableBody>
                        {ctx().weeks.map((week) => (
                          <DatePickerBase.TableRow>
                            {week.map((day) => (
                              <DatePickerBase.TableCell value={day}>
                                <DatePickerBase.TableCellTrigger class={buttonVariants({ variant: "ghost" })}>
                                  {day.day}
                                </DatePickerBase.TableCellTrigger>
                              </DatePickerBase.TableCell>
                            ))}
                          </DatePickerBase.TableRow>
                        ))}
                      </DatePickerBase.TableBody>
                    </DatePickerBase.Table>
                  </>
                )}
              </DatePickerBase.Context>
            </DatePickerBase.View>
            {/* Month and Year views omitted for brevity — see source */}
          </DatePickerBase.Content>
        </DatePickerBase.Positioner>
      </Portal>
    </DatePickerBase.Root>
  );
};

const DatePickerLabel: Component<ArkDatePicker.LabelProps> = (props) => (
  <DatePickerBase.Label {...props} />
);

// Composite: renders styled Control + Input + calendar Trigger + ClearTrigger
const DatePickerControl: Component<ArkDatePicker.InputProps> = (props) => {
  const [local, others] = splitProps(props, ["class"]);
  return (
    <DatePickerBase.Control class={styles.control({ class: local.class })}>
      <DatePickerBase.Input {...others} />
      <DatePickerBase.Trigger>
        <CalendarIcon />
      </DatePickerBase.Trigger>
      <DatePickerBase.ClearTrigger>Clear</DatePickerBase.ClearTrigger>
    </DatePickerBase.Control>
  );
};

export { DatePicker, DatePickerLabel, DatePickerControl, DatePickerBase };
export { datePickerVariants, type DatePickerVariants } from "../../recipes/date-picker";
```

## Usage

Import the components:

```tsx
import {
  DatePicker,
  DatePickerLabel,
  DatePickerControl,
  DatePickerRootProvider,
  DatePickerYearSelect,
  DatePickerMonthSelect,
} from "~/components/date-picker";

// For custom composition, import DatePickerBase:
import { DatePickerBase } from "~/components/date-picker/date-picker.base";
```

Basic usage:

```tsx
// DatePickerControl renders a styled Control with Input + calendar Trigger + ClearTrigger.
<DatePicker>
  <DatePickerLabel>Label</DatePickerLabel>
  <DatePickerControl />
</DatePicker>
```

## Controlled

Control the selected date programmatically using `value` and `onValueChange`.

```tsx
import { createSignal } from "solid-js";
import {
  DatePicker,
  DatePickerLabel,
  DatePickerControl,
  DatePickerRootProvider,
  DatePickerYearSelect,
  DatePickerMonthSelect,
} from "~/components/date-picker";
import { CalendarDate } from "@internationalized/date";

export function DatePickerControlled() {
  const [value, setValue] = createSignal<CalendarDate[]>([]);

  return (
    <DatePicker value={value()} onValueChange={(details) => setValue(details.value)}>
      <DatePickerLabel>Select date</DatePickerLabel>
      <DatePickerControl />
    </DatePicker>
  );
}
```
