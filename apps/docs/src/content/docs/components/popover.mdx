import { Popover as ArkPopover } from '@ark-ui/solid/popover'

import { createMemo, splitProps, type Component, type JSX } from 'solid-js'
import { popoverVariants } from './recipes/popover'

// Global variant instance (no params consumed)
const styles = popoverVariants()

export const PopoverRoot = ArkPopover.Root;
export const PopoverRootProvider = ArkPopover.RootProvider;
export const PopoverTrigger = ArkPopover.Trigger;

const PopoverContent: Component<PopoverContentProps> = (props) => {
const [local, others] = splitProps(props, ['class', 'children'])
// uses global `styles`
const contentClass = createMemo(() => styles.content({ class: local.class }))
return (

<Portal>
<ArkPopover.Positioner class={styles.positioner()}>
<ArkPopover.Content class={contentClass()} {...others}>
{local.children}
</ArkPopover.Content>
</ArkPopover.Positioner>
</Portal>
)
}

## Usage

Import the components:

```tsx
import {
  Popover,
  PopoverRootProvider,
  PopoverTrigger,
  PopoverContent,
  PopoverTitle,
  PopoverDescription,
} from "~/components/popover";
```

Basic usage:

```tsx
<Popover>
  <PopoverTrigger as={Button}>Open Popover</PopoverTrigger>
  <PopoverContent>
    <PopoverTitle>Title</PopoverTitle>
    <PopoverDescription>Description text</PopoverDescription>
  </PopoverContent>
</Popover>
```

## With Close Trigger

Add a close trigger to allow users to dismiss the popover.

```tsx
<Popover>
  <PopoverTrigger as={Button}>Open</PopoverTrigger>
  <PopoverContent>
    <PopoverTitle>Title</PopoverTitle>
    <PopoverDescription>Description</PopoverDescription>
    <PopoverCloseTrigger>Close</PopoverCloseTrigger>
  </PopoverContent>
</Popover>
```

## Root Provider

Use `PopoverRootProvider` when you need to access the popover state outside of the component tree. This pattern uses the `usePopover` hook from Ark UI to create a shared context that both the popover and external elements can reference.

<PopoverRootProviderDemo client:only="solid-js" />

```tsx
import { usePopover } from "@ark-ui/solid/popover";
import {
  PopoverRootProvider,
  PopoverTrigger,
  PopoverContent,
  PopoverTitle,
  PopoverDescription,
  PopoverCloseTrigger,
} from "~/components/popover";
import { Button } from "~/components/button";

export function PopoverWithExternalControl() {
  const popover = usePopover();

  return (
    <div>
      {/* Access popover state outside the tree */}
      <output>Open: {JSON.stringify(popover().open)}</output>

      <PopoverRootProvider value={popover}>
        <PopoverTrigger as={Button}>Open Popover</PopoverTrigger>
        <PopoverContent>
          <PopoverTitle>Popover Title</PopoverTitle>
          <PopoverDescription>
            This popover state is managed externally via <code>usePopover</code>.
          </PopoverDescription>
          <PopoverCloseTrigger>Close</PopoverCloseTrigger>
        </PopoverContent>
      </PopoverRootProvider>
    </div>
  );
}
```

The key difference:

- **`Popover` (Root)** — manages its own state internally. Use for simple, self-contained popovers.
- **`PopoverRootProvider`** — accepts a pre-created popover context via `usePopover`. Use when you need to read or control the popover state from outside the component tree.

## With Arrow

Add an arrow to visually connect the popover to its trigger.

```tsx
<Popover>
  <PopoverTrigger as={Button}>Open</PopoverTrigger>
  <PopoverContent>
    <PopoverArrow>
      <PopoverArrowTip />
    </PopoverArrow>
    <PopoverTitle>Title</PopoverTitle>
    <PopoverDescription>Description</PopoverDescription>
  </PopoverContent>
</Popover>
```

## API Reference

### Popover (Root)

| Prop         | Type                      | Default | Description                          |
| ------------ | ------------------------- | ------- | ------------------------------------ |
| open         | `boolean`                 | —       | Controlled open state.               |
| onOpenChange | `(open: boolean) => void` | —       | Callback when open state changes.    |
| positioning  | `object`                  | —       | Positioning options for the popover. |
| class        | `string`                  | —       | Custom CSS class.                    |

### PopoverRootProvider

| Prop  | Type               | Default | Description                                               |
| ----- | ------------------ | ------- | --------------------------------------------------------- |
| value | `UsePopoverReturn` | —       | Popover context returned by `usePopover()`. **Required.** |
| class | `string`           | —       | Custom CSS class.                                         |

### PopoverTrigger

| Prop    | Type      | Default | Description                                |
| ------- | --------- | ------- | ------------------------------------------ |
