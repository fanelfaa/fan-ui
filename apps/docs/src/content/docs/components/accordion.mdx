## Installation

### CLI
Run the following command to add the component to your project:
```bash
npx solidui-cli@latest add accordion
```

### Manual
<div class="space-y-3">
Install the dependency:
```bash
npm install tailwind-variants
```
</div>
<div class="space-y-3">
Create the recipe file at `src/components/recipes/accordion.ts`:
```tsx
import { tv, type VariantProps } from 'tailwind-variants'

export const accordionVariants = tv({
  slots: {
    root: 'w-full',
    item: 'border-b border-border',
    itemTrigger:
      'flex w-full items-center justify-between gap-2 py-4 text-sm font-medium ring-offset-background transition-all focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-ring focus-visible:ring-offset-2 disabled:pointer-events-none disabled:opacity-50',
    itemIndicator: 'size-4 shrink-0 text-foreground transition-transform duration-200 data-[state=open]:rotate-180',
    itemContent:
      'overflow-hidden transition-all duration-200 data-[state=open]:animate-in data-[state=closed]:animate-out data-[state=open]:fade-in-0 data-[state=closed]:fade-out-0',
    itemBody: 'pb-4 text-sm text-foreground',
  },
})

export type AccordionVariants = VariantProps<typeof accordionVariants>
```
</div>
<div class="space-y-3">
Create the component file at `src/components/accordion.tsx`:
```tsx
import { Accordion as ArkAccordion } from '@ark-ui/solid/accordion'
import { createMemo, splitProps, type Component } from 'solid-js'
import { accordionVariants } from './recipes/accordion'

const AccordionRoot: Component<ArkAccordion.RootProps> = (props) => {
  const [local, others] = splitProps(props, ['class'])
  const styles = accordionVariants()
  const rootClass = createMemo(() => styles.root({ class: local.class }))
  return <ArkAccordion.Root class={rootClass()} {...others} />
}

const AccordionItem: Component<ArkAccordion.ItemProps> = (props) => {
  const [local, others] = splitProps(props, ['class'])
  const styles = accordionVariants()
  const itemClass = createMemo(() => styles.item({ class: local.class }))
  return <ArkAccordion.Item class={itemClass()} {...others} />
}

const AccordionItemTrigger: Component<ArkAccordion.ItemTriggerProps> = (props) => {
  const [local, others] = splitProps(props, ['class'])
  const styles = accordionVariants()
  const triggerClass = createMemo(() => styles.itemTrigger({ class: local.class }))
  return <ArkAccordion.ItemTrigger class={triggerClass()} {...others} />
}

const AccordionItemContent: Component<ArkAccordion.ItemContentProps> = (props) => {
  const [local, others] = splitProps(props, ['class'])
  const styles = accordionVariants()
  const contentClass = createMemo(() => styles.itemContent({ class: local.class }))
  return <ArkAccordion.ItemContent class={contentClass()} {...others} />
}

const AccordionItemIndicator: Component<ArkAccordion.ItemIndicatorProps> = (props) => {
  const [local, others] = splitProps(props, ['class'])
  const styles = accordionVariants()
  const indicatorClass = createMemo(() => styles.itemIndicator({ class: local.class }))
  return <ArkAccordion.ItemIndicator class={indicatorClass()} {...others} />
}

export {
  AccordionRoot as Accordion,
  AccordionItem,
  AccordionItemTrigger,
  AccordionItemContent,
  AccordionItemIndicator,
  accordionVariants,
}
```
</div>
> **Note:** Make sure your project has the Tailwind CSS theme variables set up (`--ring`, `--border`, `--background`, etc.) or override the utility classes to match your design system.

## Usage

Import the components:

```tsx
import {
  Accordion,
  AccordionItem,
  AccordionItemTrigger,
  AccordionItemContent,
  AccordionItemIndicator,
} from "~/components/accordion"
```

Basic usage:

```tsx
<Accordion>
  <AccordionItem value="item-1">
    <AccordionItemTrigger>
      Section Title
      <AccordionItemIndicator>
        <ChevronDownIcon />
      </AccordionItemIndicator>
    </AccordionItemTrigger>
    <AccordionItemContent>
      <div class="pb-4 text-sm">Content goes here.</div>
    </AccordionItemContent>
  </AccordionItem>
</Accordion>
```

With multiple items:

```tsx
<Accordion defaultValue={["item-1"]}>
  <AccordionItem value="item-1">
    <AccordionItemTrigger>
      First Section
      <AccordionItemIndicator><ChevronDownIcon /></AccordionItemIndicator>
    </AccordionItemTrigger>
    <AccordionItemContent>
      <div class="pb-4 text-sm">First content.</div>
    </AccordionItemContent>
  </AccordionItem>
  <AccordionItem value="item-2">
    <AccordionItemTrigger>
      Second Section
      <AccordionItemIndicator><ChevronDownIcon /></AccordionItemIndicator>
    </AccordionItemTrigger>
    <AccordionItemContent>
      <div class="pb-4 text-sm">Second content.</div>
    </AccordionItemContent>
  </AccordionItem>
</Accordion>
```

## Multiple

Use the `multiple` prop to allow more than one item to be expanded at the same time.

<AccordionMultipleDemo client:load />

```tsx
<Accordion multiple defaultValue={["item-1", "item-2"]}>
  <AccordionItem value="item-1">
    <AccordionItemTrigger>
      Can I open multiple items?
      <AccordionItemIndicator><ChevronDownIcon /></AccordionItemIndicator>
    </AccordionItemTrigger>
    <AccordionItemContent>
      <div class="pb-4 text-sm">
        Yes. Just pass the <code>multiple</code> prop.
      </div>
    </AccordionItemContent>
  </AccordionItem>
  {/* ... more items */}
</Accordion>
```

## Collapsible

Use the `collapsible` prop to allow items to be closed after they have been expanded. By default, at least one item stays open.

```tsx
      <div class="pb-4 text-sm">
        Yes. Click the trigger again to collapse.
      </div>
    </AccordionItemContent>
  </AccordionItem>
</Accordion>
```

## Controlled

Control the expanded state programmatically using `value` and `onValueChange`.

```tsx
import { createSignal } from "solid-js"
import {
  Accordion,
  AccordionItem,
  AccordionItemTrigger,
  AccordionItemContent,
  AccordionItemIndicator,
} from "~/components/accordion"

export function AccordionControlled() {
  const [value, setValue] = createSignal<string[]>([])

  return (
    <Accordion
      value={value()}
      onValueChange={(details) => setValue(details.value)}
    >
      <AccordionItem value="item-1">
        <AccordionItemTrigger>
          Controlled Item
          <AccordionItemIndicator><ChevronDownIcon /></AccordionItemIndicator>
        </AccordionItemTrigger>
        <AccordionItemContent>
          <div class="pb-4 text-sm">
            Current state: {value().join(", ") || "closed"}
          </div>
        </AccordionItemContent>
      </AccordionItem>
    </Accordion>
  )
}
```

## Disabled

Disable individual items or the entire accordion.

<AccordionDisabledDemo client:load/>

```tsx
<Accordion>
  <AccordionItem value="item-1">
    <AccordionItemTrigger>Active Item</AccordionItemTrigger>
    <AccordionItemContent>
      <div class="pb-4 text-sm">This item is interactive.</div>
    </AccordionItemContent>
  </AccordionItem>
  <AccordionItem value="item-2" disabled>
    <AccordionItemTrigger>Disabled Item</AccordionItemTrigger>
    <AccordionItemContent>
      <div class="pb-4 text-sm">This item is disabled.</div>
    </AccordionItemContent>
  </AccordionItem>
</Accordion>
```

To disable all items at once, pass `disabled` to the `<Accordion>` root.

## API Reference

### Accordion (Root)

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| defaultValue | `string[]` | `[]` | Initial expanded items. Use when uncontrolled. |
| value | `string[]` | — | Controlled value of expanded items. |
| onValueChange | `(details: { value: string[] }) => void` | — | Callback when expand state changes. |
| multiple | `boolean` | `false` | Allow multiple items expanded simultaneously. |
| collapsible | `boolean` | `false` | Allow items to be closed after expansion. |
| disabled | `boolean` | `false` | Disable all accordion items. |
| orientation | `"horizontal"` \| `"vertical"` | `"vertical"` | Layout orientation of the accordion. |
| lazyMount | `boolean` | `false` | Enable lazy mounting of content. |
| unmountOnExit | `boolean` | `false` | Unmount content on exit animation. |
| class | `string` | — | Custom CSS class. |

### AccordionItem

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| value | `string` | — | Unique identifier for the item. **Required.** |
| disabled | `boolean` | `false` | Whether this item is disabled. |
| class | `string` | — | Custom CSS class. |

### AccordionItemTrigger

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| class | `string` | — | Custom CSS class. |

### AccordionItemContent

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| class | `string` | — | Custom CSS class. |

### AccordionItemIndicator

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| class | `string` | — | Custom CSS class. |
