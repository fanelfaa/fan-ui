## Installation

### CLI
Run the following command to add the component to your project:
```bash
npx solidui-cli@latest add input
```

### Manual
<div class="space-y-3">
Install the dependency:
```bash
npm install tailwind-variants
```
</div>
<div class="space-y-3">
Create the recipe file at `src/components/recipes/input.ts`:
```tsx
import { tv, type VariantProps } from 'tailwind-variants'

export const inputVariants = tv({
  slots: {
    root: 'grid gap-1.5',
    label: 'text-sm font-medium text-foreground',
    input:
      'flex h-8 w-full rounded-md border border-input bg-background px-2.5 py-1.5 text-sm ring-offset-background file:border-0 file:bg-transparent file:text-sm file:font-medium placeholder:text-muted-foreground focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-ring focus-visible:ring-offset-2 disabled:cursor-not-allowed disabled:opacity-50',
    description: 'text-sm text-muted-foreground',
    error: 'text-sm text-destructive',
  },
  variants: {
    error: {
      true: {
        input: 'border-destructive focus-visible:ring-destructive',
      },
    },
  },
  defaultVariants: {
    error: false,
  },
})

export type InputVariants = VariantProps<typeof inputVariants>
```
</div>
<div class="space-y-3">
Create the component file at `src/components/input.tsx`:
```tsx
import { Field } from '@ark-ui/solid/field'
import { createMemo, splitProps, type Component, type JSX } from 'solid-js'
import { inputVariants } from '../recipes/input'

// Global variant instance (no params)
const styles = inputVariants()

type InputProps = {
  label?: string
  description?: string
  error?: string
  class?: string
} & JSX.IntrinsicElements['input']

const Input: Component<InputProps> = (props) => {
  const [local, others] = splitProps(props, ['label', 'description', 'error', 'class'])
  const rootClass = createMemo(() => styles.root({ class: local.class, error: !!local.error }))
  return (
    <Field.Root class={rootClass()} invalid={!!local.error}>
      {local.label && <Field.Label class={styles.label()}>{local.label}</Field.Label>}
      <Field.Input class={styles.input()} {...others} />
      {local.description && !local.error && (
        <Field.HelperText class={styles.description()}>{local.description}</Field.HelperText>
      )}
      <Field.ErrorText class={styles.error()}>{local.error}</Field.ErrorText>
    </Field.Root>
  )
}

export { Input, inputVariants }
```
</div>


## Usage

Import the component:

```tsx
import { Input } from "~/components/input"
```

Basic input:

```tsx
<Input placeholder="Enter text..." />
```

With label:

```tsx
<Input label="Username" placeholder="Enter username" />
```

With description:

```tsx
<Input label="Email" description="We'll never share your email." placeholder="email@example.com" />
```

With error state:

```tsx
<Input label="Password" error="Password is required" type="password" />
```
